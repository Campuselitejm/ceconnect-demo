create or replace function public.send_announcement(body text) returns jsonb language plpgsql security definer set search_path = public as $$ declare viewer uuid := auth.uid(); is_v boolean; is_vsc boolean; is_adm boolean; used_today int; bal int; charged int := 0; begin if body is null or length(trim(body)) = 0 then return jsonb_build_object('error','empty'); end if; if char_length(body) > 120 then return jsonb_build_object('error','too_long'); end if; if body ~* '(https?://|www\.|\.com|\.net|\.org|\.io|\.ly)' then return jsonb_build_object('error','no_links'); end if; select coalesce((badges->>'varsity')::boolean,false), coalesce((badges->>'varsity_sc')::boolean,false), (role = 'admin') into is_v, is_vsc, is_adm from profiles where id = viewer; if not (is_v or is_vsc or is_adm) then return jsonb_build_object('error','not_varsity'); end if; if is_adm then charged := 0; -- admin: always free, no cap elsif is_vsc then select count(*) into used_today from activity_events where actor_id = viewer and type = 'announce_post' and created_at >= date_trunc('day', now()); if used_today >= 2 then return jsonb_build_object('error','daily_limit'); end if; else select points into bal from user_private where user_id = viewer for update; if bal < 500 then return jsonb_build_object('error','insufficient'); end if; update user_private set points = points - 500, updated_at = now() where user_id = viewer; charged := 500; end if; insert into activity_events (actor_id, type, payload) values (viewer, 'announce_post', jsonb_build_object('text', trim(body), 'charged', charged)); return jsonb_build_object('ok', true, 'charged', charged); end; $$;
